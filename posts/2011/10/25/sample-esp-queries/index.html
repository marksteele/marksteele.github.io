select 
  tc.name as metric,
  tc.uuid as uuid,
  tc.noit as noit,
  tc.time as time,
  tc.value as value,
  tc.status as status,
  cds.name as check,
  cds.module as 
  module,cds.prefix as prefix,
  cds.target as target 
from 
  ThresholdChange.std:groupwin(uuid,name).win:length(2) tc, 
  CheckDetails cds 
where 
  cds.uuid = tc.uuid and tc.status <> prev(tc.status)
